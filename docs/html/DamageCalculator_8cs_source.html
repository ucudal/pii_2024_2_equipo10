121 public static double GetEffectivness(Type type, List<Type> types)
122 {
123 foreach (Type type1 in types)
124 {
125 Tuple<Type, Type> tuple = new Tuple<Type, Type>(type, type1);
126 if (EffectivnessDataBase.ContainsKey(tuple))
127 {
128 double effectivness = EffectivnessDataBase[tuple];
129 return effectivness;
130 }
131 else
132 {
133 return 1.0;
134 }
135 }
136
137 return 1.0;
138 }

151public static double CalculateDamage(Pokemon attackerPokemon, Pokemon attackedPokemon, Attack move)
